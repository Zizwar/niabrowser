Data Extraction

You say: "Extract all product prices and names from this e-commerce page"
// AI generates: const products = []; document.querySelectorAll('.product-item').forEach(item => { const name = item.querySelector('.product-name')?.textContent; const price = item.querySelector('.price')?.textContent; if (name && price) { products.push({ name: name.trim(), price: price.trim() }); } }); console.table(products); console.log(` Extracted ${products.length} products`);

Dark Mode

You say: "Convert this page to dark mode with smooth transition"
// AI generates: const darkModeCSS = ` * { transition: background-color 0.3s, color 0.3s !important; } body { background-color: #1a1a1a !important; color: #e0e0e0 !important; } div, section { background-color: #2d2d2d !important; } a { color: #4a90e2 !important; } `; const style = document.createElement('style'); style.textContent = darkModeCSS; document.head.appendChild(style); console.log(' Dark mode activated!');

Media Downloader

You say: "Add download buttons to all images and videos on this page"
// AI generates: document.querySelectorAll('img, video').forEach((media, index) => { const downloadBtn = document.createElement('button'); downloadBtn.textContent = ' Download'; downloadBtn.style.cssText = 'position:absolute; z-index:9999; background:#4a90e2; color:white; border:none; padding:5px 10px; border-radius:5px; cursor:pointer;'; downloadBtn.onclick = () => { const link = document.createElement('a'); link.href = media.src; link.download = `media_${index}`; link.click(); }; media.style.position = 'relative'; media.parentNode.insertBefore(downloadBtn, media.nextSibling); }); console.log(' Download buttons added!');
